Instance regP_nat : encodableP nat.
Proof.
  evar (c:nat).
  exists c.
  eexists _.
  eapply computesTime_timeLeq.
  2:now apply term_nat_enc.
  repeat intro. split. 2:easy.
  cbn [fst]. rewrite -> size_nat_enc. [c]:exact 14. unfold c, c__natsizeS, c__natsizeO. nia.
Qed.
